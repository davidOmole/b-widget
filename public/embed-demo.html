Booking Widget Demo

Inline Embed

This widget is embedded directly in the page:

<script src="https://bookings-widget-domain.com/embed.js"></script>
<script>
    initBookingWidget({
        type: 'inline',
        target: document.getElementById('inline-widget'),
        serviceId: 'your-service-id'
    });
</script>

Popup Embed

This widget opens in a popup when clicked:

<script src="https://bookings-widget-domain.com/embed.js"></script>
<script>
    initBookingWidget({
        type: 'popup',
        target: document.getElementById('popup-widget'),
        serviceId: 'your-service-id',
        buttonText: 'Book Now',
        onSuccess: (data) => {
            console.log('Booking successful:', data);
        },
        onError: (error) => {
            console.error('Booking failed:', error);
        }
    });
</script>

Popup Text Embed

This widget shows a text button that opens the booking widget in a popup:

<script src="https://bookings-widget-domain.com/embed.js"></script>
<script>
    initBookingWidget({
        type: 'popup_text',
        target: document.getElementById('popup-text-widget'),
        serviceId: 'your-service-id',
        buttonText: 'Schedule a Meeting'
    });
</script>
